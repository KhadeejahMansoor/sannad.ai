import { pool } from '../../../lib/db';
import { NextResponse } from 'next/server';

export async function GET(request) {
  try {
    const { searchParams } = new URL(request.url);

    const language = searchParams.get('lang') || 'en';
    const compiler = searchParams.get('compiler');
    const grade = searchParams.get('grade');
    const chapter = searchParams.get('chapter');
    const section = searchParams.get('section');
    const limit = parseInt(searchParams.get('limit')) || 20;
    const offset = parseInt(searchParams.get('offset')) || 0;

    // Pick the language-appropriate text column
    // Arabic = full hadith in Arabic; English = post-clause English translation
    const textColumn = language === 'ar' ? 'final_hadith' : 'post_clause_english';

    // Build the unified subquery that combines both collections.
    // Each table is normalized to the same column shape so they can UNION cleanly.
    const baseQuery = `
      SELECT
        'azami-' || id::text AS id,
        hadith_number,
        compiler,
        volume,
        book,
        section,
        chapter,
        grade,
        commentary,
        hadith_sources AS reference,
        ${textColumn} AS hadith_text,
        NULL::boolean AS is_verified,
        'azami' AS source
      FROM azami_hadiths
      WHERE ${textColumn} IS NOT NULL AND TRIM(${textColumn}) != ''

      UNION ALL

      SELECT
        'sevenbooks-' || id::text AS id,
        hadith_number,
        compiler,
        volume,
        book,
        NULL AS section,
        chapter,
        final_grade AS grade,
        commentary_darussalam AS commentary,
        NULL AS reference,
        ${textColumn} AS hadith_text,
        NULL::boolean AS is_verified,
        'sevenbooks' AS source
      FROM sevenbooks_hadiths
      WHERE ${textColumn} IS NOT NULL AND TRIM(${textColumn}) != ''
    `;

    // Wrap the union and apply filters
    let query = `SELECT * FROM (${baseQuery}) AS combined WHERE 1=1`;
    const params = [];
    let paramCount = 0;

    if (compiler) {
      paramCount++;
      query += ` AND LOWER(compiler) = LOWER($${paramCount})`;
      params.push(compiler);
    }

    if (grade) {
      paramCount++;
      query += ` AND LOWER(grade) = LOWER($${paramCount})`;
      params.push(grade);
    }

    if (chapter) {
      paramCount++;
      query += ` AND LOWER(chapter) LIKE LOWER($${paramCount})`;
      params.push(`%${chapter}%`);
    }

    if (section) {
      paramCount++;
      query += ` AND LOWER(section) LIKE LOWER($${paramCount})`;
      params.push(`%${section}%`);
    }

    // Order and paginate
    query += ` ORDER BY hadith_number LIMIT $${paramCount + 1} OFFSET $${paramCount + 2}`;
    params.push(limit, offset);

    const result = await pool.query(query, params);

    // Count query (same filters, no limit/offset)
    let countQuery = `SELECT COUNT(*) as total FROM (${baseQuery}) AS combined WHERE 1=1`;
    const countParams = [];
    let countParamCount = 0;

    if (compiler) {
      countParamCount++;
      countQuery += ` AND LOWER(compiler) = LOWER($${countParamCount})`;
      countParams.push(compiler);
    }
    if (grade) {
      countParamCount++;
      countQuery += ` AND LOWER(grade) = LOWER($${countParamCount})`;
      countParams.push(grade);
    }
    if (chapter) {
      countParamCount++;
      countQuery += ` AND LOWER(chapter) LIKE LOWER($${countParamCount})`;
      countParams.push(`%${chapter}%`);
    }
    if (section) {
      countParamCount++;
      countQuery += ` AND LOWER(section) LIKE LOWER($${countParamCount})`;
      countParams.push(`%${section}%`);
    }

    const countResult = await pool.query(countQuery, countParams);
    const total = parseInt(countResult.rows[0].total);

    return NextResponse.json({
      success: true,
      data: result.rows,
      pagination: {
        total,
        limit,
        offset,
        hasMore: offset + limit < total
      }
    });
  } catch (error) {
    console.error('Database error:', error);
    return NextResponse.json(
      { success: false, error: 'Failed to fetch hadiths', details: error.message },
      { status: 500 }
    );
  }
}

export async function GET(request) {

    const searchQuery = (searchParams.get('q') || '').trim();
    const language = searchParams.get('lang') || 'en';

    // Empty / missing â†’ empty array (= no filter on that column).
    const compilerCsv = (searchParams.get('compiler') || '').trim();
    const gradeCsv = (searchParams.get('grade') || '').trim();
    const compilers = compilerCsv ? compilerCsv.split(',').map(s => s.trim()).filter(Boolean) : [];
    const grades    = gradeCsv    ? gradeCsv.split(',').map(s => s.trim()).filter(Boolean)    : [];

    const hasText = searchQuery.length > 0;
    const hasFilters = compilers.length > 0 || grades.length > 0;

    // Need EITHER text OR filters. Empty everything â†’ empty results (don't dump the whole DB).
    if (!hasText && !hasFilters) {
      return NextResponse.json({ success: true, data: [] });
    }

    const textColumn = language === 'ar' ? 'final_hadith' : 'post_clause_english';

    // â”€â”€ Build the inner UNION subquery. If there's a text query, filter
    //    inside each table for index efficiency. Otherwise leave it open.
    const innerWhere = hasText ? `WHERE a.${textColumn} ILIKE $1` : '';
    const innerWhereSeven = hasText ? `WHERE s.${textColumn} ILIKE $1` : '';

    const baseQuery = `
      SELECT
        'azami-' || a.id::text AS hadith_id,
        a.hadith_number,
        a.compiler,
        a.volume,
        a.book,
        a.section,
        a.chapter,
        a.grade,
        'Azami'::text AS final_grader,
        a.commentary,
        a.hadith_sources AS reference,
        NULL::text AS ayat,
        a.duplicates,

        a.final_hadith AS hadith_text_arabic,

        a.intro_clause AS arabic_intro_clause,
        m.english_name AS english_narrator,
        'azami' AS source
      FROM azami_hadiths a
      LEFT JOIN machine_clause_translations m ON a.machine_clause = m.machine_clause
      ${innerWhere}

      UNION ALL

      SELECT
        'sevenbooks-' || s.id::text AS hadith_id,
        s.hadith_number,
        s.compiler,
        s.volume,
        s.book,
        NULL AS section,
        s.chapter,

        s.intro_clause AS arabic_intro_clause,

// table the hadiths live in. sevenbooks_hadiths has no `section` column so
// we expose NULL there; only azami rows actually contribute sections today.
export async function GET(request) {
  try {
    const { searchParams } = new URL(request.url);
    const compiler = searchParams.get('compiler');
    const book     = searchParams.get('book');
    const chapter  = searchParams.get('chapter');

    if (!compiler || !book || !chapter) {
      return NextResponse.json(
        { success: false, error: 'compiler, book, and chapter parameters are all required' },
        { status: 400 }
      );
    }

    const result = await pool.query(`
      SELECT DISTINCT section FROM (
        SELECT section FROM azami_hadiths
        WHERE compiler = $1 AND book = $2 AND chapter = $3
          AND section IS NOT NULL AND TRIM(section) != ''
        UNION
        SELECT NULL AS section FROM sevenbooks_hadiths
        WHERE compiler = $1 AND book = $2 AND chapter = $3
          AND FALSE  -- sevenbooks has no sections; keep the UNION shape consistent
      ) AS combined
      WHERE section IS NOT NULL
      ORDER BY section
    `, [compiler, book, chapter]);

    return NextResponse.json({
      success: true,
      data: result.rows.map(r => r.section),
    });
  } catch (error) {
    console.error('Database error:', error);
    return NextResponse.json(
      { success: false, error: 'Failed to fetch sections', details: error.message },
      { status: 500 }
    );
  }
}

import { pool } from '../../../lib/db';
import { NextResponse } from 'next/server';

export async function GET(request) {
  try {
    const { searchParams } = new URL(request.url);
    const term = searchParams.get('term') || '';
    const language = searchParams.get('lang') || 'en';

    if (term.trim().length < 2) {
      return NextResponse.json({ success: true, data: [] });
    }

    const textColumn = language === 'ar' ? 'final_hadith' : 'post_clause_english';
    const likePattern = `%${term}%`;

    // Pull a few snippets from each table where the search column matches.
    const result = await pool.query(`
      SELECT DISTINCT suggestion FROM (
        SELECT ${textColumn} AS suggestion FROM azami_hadiths
        WHERE ${textColumn} ILIKE $1 LIMIT 5

        UNION

        SELECT ${textColumn} AS suggestion FROM sevenbooks_hadiths
        WHERE ${textColumn} ILIKE $1 LIMIT 5
      ) AS combined
      LIMIT 10
    `, [likePattern]);

    return NextResponse.json({
      success: true,
      data: result.rows
    });
  } catch (error) {
    console.error('Suggestions error:', error);
    return NextResponse.json(
      { success: false, error: 'Failed to fetch suggestions' },
      { status: 500 }
    );
  }
}
